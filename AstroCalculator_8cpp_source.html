44 Fittino::AstroCalculator::AstroCalculator( const ModelBase* model, const boost::property_tree::ptree& ptree )
45  : CalculatorBase( model ),
46  _simpleOutputDataStorage( new SimpleDataStorage() ) {
47 
48  _name = "AstroCalculator";
49  if( ptree.count("Chi2Contribution" ) != 0 ) {
50  BOOST_FOREACH( const boost::property_tree::ptree::value_type & node, ptree ) { //.get_child("Chi2Contribution") ) {
51  if( node.first != "Chi2Contribution" ) continue;
52  std::string name = node.second.get<std::string>("Name");
53  _chi2ContributionNames.push_back( name );
54  _simpleOutputDataStorage -> AddEntry( name, 0. );
55  _xValueNames.push_back( node.second.get<std::string>( "XValue" ) );
56  _yValueNames.push_back( node.second.get<std::string>( "YValue" ) );
57  _dataPointFileNames.push_back( node.second.get<std::string>( "FileName" ) );
58  _exclusionGraphs.push_back( TGraph(_dataPointFileNames.at(_dataPointFileNames.size()-1).c_str()) );
59  _theoryUncertainties.push_back( node.second.get<double>( "TheoryUncertainty" ) );
60  _widthFormulas.push_back( TF1("", (node.second.get<std::string>( "WidthFormula" ) ).c_str() ) );
61  _limitFormulas.push_back( TF1("", (node.second.get<std::string>( "LimitFormula" ) ).c_str() ) );
62  _measuredValues.push_back( 0. );
63  AddQuantity( new SimplePrediction( name, "", _simpleOutputDataStorage -> GetMap() -> at(name) ) );
64 
65  }
66  }

75 void Fittino::AstroCalculator::CalculatePredictions() {
76 
77  for( unsigned int i = 0; i < GetCollectionOfQuantities().GetNumberOfElements(); ++i ) {
78 
79  std::string name = GetCollectionOfQuantities().At(i) -> GetName();
80  double xValue = _model -> GetCollectionOfQuantities().At( _xValueNames.at(i) ) -> GetValue();
81  double yValue = _model -> GetCollectionOfQuantities().At( _yValueNames.at(i) ) -> GetValue();
82  _widthFormulas.at(i).SetParameter( 0, _exclusionGraphs.at(i).Eval( xValue ) );
83  _widthFormulas.at(i).SetParameter( 1, _theoryUncertainties.at(i) );
84  _widthFormulas.at(i).SetParameter( 2, yValue );
85  _limitFormulas.at(i).SetParameter( 0, _measuredValues.at(i) );
86  _limitFormulas.at(i).SetParameter( 1, _widthFormulas.at(i).Eval(0) );
87  double width = _widthFormulas.at(i).Eval(0);
88  double csLimit = width*TMath::Sqrt(_limitFormulas.at(i).Eval(0));
89 
90  if( yValue > csLimit ) {
91  _simpleOutputDataStorage -> GetMap() -> at( name ) = ( yValue - csLimit )*(yValue - csLimit)/width/width;
92  }
93  else {
94  _simpleOutputDataStorage -> GetMap() -> at( name ) = 0.;
95  }
96 
97  }
