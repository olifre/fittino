51 void Fittino::CheckMATE2Calculator::ReadResult() {
52 
53  std::string line;
54  std::ifstream file( _directory + "/result.txt" );
55  if( !file.is_open() ) throw LogicException( "Cannot open result.txt." );
56 
57  getline( file, line );
58  if ( ! line.empty() ) throw LogicException( "CheckMATECalculator: Expected empty line. " );
59 
60  getline( file, line );
61  std::string clsid = "Test: Calculation of CLs from determined signal";
62  std::string rid = "Test: Calculation of CLs from determined signal";
63 
64  if( _first && line == clsid ) _fullCLs = true;
65  else if( _first && line==rid ) _fullCLs = false;
66  else if( _first ) throw LogicException( "CheckMATECalculator: Expected Test line." );
67 
68  if( _fullCLs && line != clsid ) throw LogicException( "CheckMATECalculator: Expected CLs test." );
69  else if ( !_fullCLs && line !=rid ) throw LogicException( "Expected r test." );
70 
71  getline( file, line );
72  _errorIsDominatedByMonteCarloStatistics = 0;
73  _theModelCouldBeExcludedIfYouProvidedMoreInputEvents = 0;
74 
75  while( boost::starts_with( line, "Warning:" ) ) {
76 
77  if ( line == "Warning: Error is dominated by Monte Carlo statistics!" ) _errorIsDominatedByMonteCarloStatistics = 1;
78  else if ( line == "Warning: The model could be excluded if you provided more input events!" ) _theModelCouldBeExcludedIfYouProvidedMoreInputEvents = 1;
79  else throw LogicException( "CheckMATE2Calculator: Unknown warning: " + line );
80  getline( file, line );
81 
82  }
83 
84  if ( line == "Result: Excluded" ) _excluded = 1;
85  else if ( line == "Result: Allowed" ) _excluded = 0;
86  else throw LogicException( "Expected result." );
87 
88  getline( file, line );
89  clsid = "Result for CLs: cls_min = ";
90  rid = "Result for r: r_max = ";
91 
92  if ( _fullCLs ) {
93 
94  if ( !boost::starts_with( line, clsid ) ) throw LogicException( "CheckMATECalculator: Expected CLs in result.txt." );
95  boost::algorithm::erase_first( line, clsid );
96  _cls = std::stod( line );
97 
98  getline( file, line );
99 
100  }
101 
102  if ( !boost::starts_with( line, rid ) ) throw LogicException( "CheckMATECalculator: Expected r in result.txt, got " + line );
103  boost::algorithm::erase_first( line, rid );
104  _r = std::stod( line );
105 
106  getline( file, line );
107  if ( !boost::starts_with( line, "SR: " ) ) throw LogicException( "Expected SR." );
108  boost::algorithm::erase_first( line, "SR: " );
109  _sr = line;
110 
111  if ( getline( file, line ) ) throw LogicException( "Expected end of result.txt." );
112 
113  file.close();
114
