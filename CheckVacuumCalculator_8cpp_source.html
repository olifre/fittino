50 void Fittino::CheckVacuumCalculator::CalculatePredictions() {
51 
52  // The Stability function takes in m_0, m_1/2, Tan(beta), sign(mu) and A_0.
53  // It outputs 1 , 0 or -1 for stable, long-lived and short-lived
54 
55  _vacuumStability = _checkVacuum->Stability( _m0, _m12, _tanbeta, _signmu, _a0 );
56 
57  // The Lifetime function takes in m_0, m_1/2, Tan(beta), sign(mu) and A_0.
58  // If the input point is short-lived it outputs the lifetime in universe ages
59  // If the input point is long-lived it outputs 1e+6
60  // If the input point is stable it outputs -1
61 
62  _vacuumLifetime = _checkVacuum->Lifetime ( _m0, _m12, _tanbeta, _signmu, _a0 );

64 }
