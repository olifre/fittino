97 void Fittino::Chi2Calculator::CalculatePredictions() {
98 
99  _lastCov = _cov;
100  _cov.Zero();
101 
102  _chi2 = 0;
103 
104  for ( unsigned int i = 0; i < _observables.size(); i++ ) {
105 
106  auto obs = _observables[i];
107 
108  obs->Update();
109 
110  _chi2 += obs->GetChi2();
111 
112  _deviations[i] = obs->GetDeviation();
113  _cov[i][i] = TMath::Power( obs->GetTotalUncertainty(), 2 );
114 
115  }
116 
117  _correlatedChi2 = _chi2;
118 
119  if ( _correlations.empty() ) return;
120 
121  for ( auto correlation : _correlations ) {
122 
123  correlation->Update();
124 
125  auto i = correlation->GetID().second.first;
126  auto j = correlation->GetID().second.second;
127 
128  _cov[i][j] += correlation->GetCovariance();
129  _cov[j][i] += _cov[i][j];
130 
131  }
132 
133  _inverted = _inverted && _cov == _lastCov;
134  if ( !_inverted ) InvertCovarianceMatrix();
135  _inverted = true;
136 
137  _deviationsTransposed = _deviations;
138  _deviationsTransposed *= _invCov;
139  _correlatedChi2 = _deviationsTransposed * _deviations;
140 
141 }
