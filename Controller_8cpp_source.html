47 void Fittino::Controller::ExecuteFittino() const {
48 
49  try {
50 
51  const Factory factory;
52  ModelBase* model = 0;
53  std::string modelType;
54 
55  if ( _inputPtree->get_child( "InputFile" ).count( "Model" ) ) {
56 
57  boost::property_tree::ptree::value_type &modelNode = *( _inputPtree->get_child( "InputFile.Model" ).begin() );
58  modelType = modelNode.first;
59  boost::property_tree::ptree &modelTree = modelNode.second;
60  model = factory.CreateModel( modelType, modelTree );
61 
62  }
63 
64  for( auto toolNode : _inputPtree->get_child( "InputFile.Tool" ) ) {
65 
66  std::string toolType = toolNode.first;
67  boost::property_tree::ptree& toolTree = toolNode.second;
68  Tool* tool = factory.CreateTool( toolType, model, toolTree );
69 
70  if ( !_lockFileName.empty() ) {
71 
72 #if Boost_MINOR_VERSION<56
73 
74  boost::property_tree::xml_writer_settings<char> settings( '\t', 1 );
75 
76 #else
77 
78  auto settings = boost::property_tree::xml_writer_make_settings<std::string> ( '\t', 1 );
79 
80 #endif
81 
82  boost::property_tree::write_xml( _inputFileName, *_inputPtree, std::locale(), settings );
83 
84  _scopedLock->unlock();
85 
86  }
87 
88  tool->PerformTask();
89 
90  _outputPtree->put( "InputFile.VerbosityLevel", _inputPtree->get<std::string>( "InputFile.VerbosityLevel" ) );
91 
92  if ( _inputPtree->get_child( "InputFile" ).count( "RandomSeed" ) ) {
93 
94  _outputPtree->put( "InputFile.RandomSeed", _inputPtree->get<std::string>( "InputFile.RandomSeed" ) );
95 
96  }
97 
98  if ( _inputPtree->get_child( "InputFile" ).count( "Model" ) ) {
99 
100  _outputPtree->put_child( "InputFile.Model." + modelType, model->GetPropertyTree() );
101 
102  }
103 
104  _outputPtree->put_child( "InputFile.Tool." + toolType , tool->GetPropertyTree() );
105 
106  delete tool;
107 
108  }
109 
110  delete model;
111 
112  }
113  catch ( const ConfigurationException& configurationException ) {
114 
115  std::cout << "\n" << configurationException.what() << "\n" << std::endl;
116  exit( EXIT_FAILURE );
117 
118  }

122 void Fittino::Controller::InitializeFittino( int argc, char** argv ) {
123 
124  try {
125 
126  // If Fittino is called without options or arguments print a help text with further
127  // instructions and exit.
128 
129  if ( argc == 1 ) {
130 
131  Controller::PrintHelp();
132  exit( EXIT_SUCCESS );
133 
134  }
135 
136  // Otherwise handle given command line options.
137 
138  Controller::HandleOptions( argc, argv );
139 
140  // Print a welcome logo.
141 
142  Controller::PrintLogo();
143 
144  // A given XML input file is parsed, and the configuration items are stored in the input
145  // property tree.
146 
147  Controller::CheckInputFileFormat();
148 
149  if ( ! _lockFileName.empty() ) {
150 
151  _fileLock = new boost::interprocess::file_lock( _lockFileName.c_str() );
152  _scopedLock = new boost::interprocess::scoped_lock<boost::interprocess::file_lock>( *_fileLock );
153 
154  }
155 
156  if ( _validate ) {
157 
158  Controller::ValidateInputFile();
159 
160  }
161 
162  boost::property_tree::read_xml( _inputFileName,
163  *_inputPtree,
164  boost::property_tree::xml_parser::trim_whitespace |
165  boost::property_tree::xml_parser::no_comments );
166 
167  // Set the verbosity level of the output text.
168 
169  std::string verbosityLevel = _inputPtree->get<std::string>( "InputFile.VerbosityLevel" );
170  Messenger::GetInstance().SetVerbosityLevel( verbosityLevel );
171 
172  if ( _inputPtree->get_child( "InputFile" ).count( "RandomSeed" ) ) {
173 
174  unsigned int randomSeed = _inputPtree->get<unsigned int>( "InputFile.RandomSeed" );
175 
176  RandomGenerator::GetInstance()->SetSeed( randomSeed );
177 
178  if ( randomSeed == 0 ) {
179 
180  Messenger::GetInstance() << Messenger::ALWAYS << "RandomSeed was set to 0 in the input file. A random random seed of " << RandomGenerator::GetInstance()->GetSeed() << " is used." << Messenger::Endl;
181 
182  }
183 
184  }
185 
186  }
187  catch ( const InputException& inputException ) {
188 
189  std::cout << "\n" << inputException.what() << "\n" << std::endl;
190  exit( EXIT_FAILURE );
191 
192  }

259 void Fittino::Controller::HandleOptions( int argc, char** argv ) {
260 
261  // Use getopt() to handle given command line options. For more informations on getopt() have a
262  // look at the manpage of getopt(3).
263 
264  static struct option options[] = {
265 
266  {"help", no_argument, 0, 'h'},
267  {"input-file", required_argument, 0, 'i'},
268  {"lock-file", required_argument, 0, 'l'},
269  {"validate", no_argument, 0, 'v'},
270  {0, 0, 0, 0 }
271 
272  };
273 
274  int optionIndex = 0;
275  int optionCode = -1;
276  opterr = 0;
277 
278  while ( true ) {
279 
280  optionCode = getopt_long( argc, argv, ":hi:l:v", options, &optionIndex );
281 
282  if ( optionCode == -1 ) break;
283 
284  switch ( optionCode ) {
285 
286  case 'h':
287  Controller::PrintHelp();
288  exit( EXIT_SUCCESS );
289 
290  case 'i':
291  _inputFileName = std::string( optarg );
292  continue;
293 
294  case 'l':
295  _lockFileName = std::string( optarg );
296  continue;
297 
298  case 'v' :
299  _validate = true;
300  continue;
301 
302  case ':':
303  throw InputException( "Missing option parameter." );
304 
305  default:
306  throw InputException( "Unknown option(s)" );
307 
308  }
309 
310  }
