59 void Fittino::EdgeDetectionTool::Execute() {
60 
61  std::vector<TH2D*> hcHistograms;
62  std::vector<TH2D*> edgeHistograms;
63 
64  for ( unsigned int i = 0; i < _histogramVector.size(); ++i ) {
65 
66  // Create the high contrast histogram.
67 
68  hcHistograms.push_back( ( ( TH2D* )( _histogramVector.at( i )->Clone() ) ) );
69 
70  for ( int binX = 0; binX <= hcHistograms.at( i )->GetNbinsX(); ++binX ) {
71 
72  for ( int binY = 0; binY <= hcHistograms.at( i )->GetNbinsY(); ++binY ) {
73 
74  if ( hcHistograms.at( i )->GetBinContent( binX, binY ) > 0. ) {
75 
76  hcHistograms.at( i )->SetBinContent( binX, binY, 1. );
77 
78  }
79  else {
80 
81  hcHistograms.at( i )->SetBinContent( binX, binY, 0. );
82 
83  }
84 
85  }
86 
87  }
88 
89  // Now create the edge hisotgrams and remove the outliers.
90 
91  edgeHistograms.push_back( ( ( TH2D* )( _histogramVector.at( i )->Clone( "" ) ) ) );
92  unsigned int cleaningStep = 0;
93 
94  while ( cleaningStep < _numberOfCleaningSteps ) {
95 
96  for ( int binX = 0; binX < hcHistograms.at( i )->GetNbinsX(); ++binX ) {
97 
98  for ( int binY = 0; binY < hcHistograms.at( i )->GetNbinsY(); ++binY ) {
99 
100  double cross = abs( hcHistograms.at( i )->GetBinContent( binX, binY )
101  - hcHistograms.at( i )->GetBinContent( binX + 1, binY + 1 ) )
102  + abs( hcHistograms.at( i )->GetBinContent( binX + 1, binY )
103  - hcHistograms.at( i )->GetBinContent( binX , binY + 1 ) );
104  edgeHistograms.at( i )->SetBinContent( binX, binY, cross );
105 
106  }
107 
108  }
109 
110  for ( int binX = 0; binX < hcHistograms.at( i )->GetNbinsX(); ++binX ) {
111 
112  for ( int binY = 0; binY < hcHistograms.at( i )->GetNbinsY(); ++binY ) {
113 
114  double difference = hcHistograms.at( i )->GetBinContent( binX, binY )
115  - edgeHistograms.at( i )->GetBinContent( binX, binY );
116  if ( difference < 0. || difference > 0. ) {
117 
118  continue;
119 
120  }
121  else {
122 
123  hcHistograms.at( i )->SetBinContent( binX, binY, 0. );
124 
125  }
126 
127  }
128 
129  }
130  ++cleaningStep;
131 
132  }
133 
134  std::vector<double> xvalues;
135  std::vector<double> yvalues;
136 
137  for ( int binX = 0; binX < hcHistograms.at( i )->GetNbinsX(); ++binX ) {
138 
139  double xval = hcHistograms.at( i )->GetXaxis()->GetBinCenter( binX );
140  double yval = hcHistograms.at( i )->GetYaxis()->GetXmin();
141 
142  for ( int binY = 0; binY < hcHistograms.at( i )->GetNbinsY(); ++binY ) {
143 
144  yval = hcHistograms.at( i )->GetYaxis()->GetBinCenter( binY );
145 
146  if ( edgeHistograms.at( i )->GetBinContent( binX, binY ) > 0. ) {
147 
148  break;
149 
150  }
151 
152  }
153 
154  xvalues.push_back( xval );
155  yvalues.push_back( yval );
156 
157  }
158 
159  unsigned int numberOfNodes = xvalues.size();
160 
161  TGraph *graph = new TGraph( numberOfNodes, &xvalues.at( 0 ), &yvalues.at( 0 ) );
162  std::string splineName = _histogramNameVector.at( i ) + "_spline";
163  std::string origSplineName = splineName + "_orig";
164  TSpline3 *spline = new TSpline3( splineName.c_str(), graph );
165  spline->SetLineColor( kRed );
166  spline->SetLineWidth( 4 );
167  spline->SetNpx( 1000 );
168 
169  _splineVector.push_back( ( ( TSpline3* )( spline->Clone( splineName.c_str() ) ) ) );
170  delete graph;
171  delete spline;
172 
173  }
