65 void Fittino::GeneticAlgorithmOptimizer::CalculateMatingProbabilities() {
66 
67  _cumulativeMatingProbabilities[0] = 0;
68  double sum = 0;
69 
70  for ( unsigned int i = 0; i < _numberOfFamilies; i++ ) {
71 
72  _matingProbabilities[i] = _numberOfFamilies - i;
73  sum = sum + _matingProbabilities[i];
74 
75  }
76 
77  for ( unsigned int i = 0; i < _numberOfFamilies; i++ ) {
78 
79  _matingProbabilities[i] = _matingProbabilities[i] / sum;
80  _cumulativeMatingProbabilities[i + 1] = _cumulativeMatingProbabilities[i] + _matingProbabilities[i];
81 
82  }

92
