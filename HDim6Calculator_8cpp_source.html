291 void Fittino::HDim6Calculator::CallFunction() {
292 
293  bool new_mh = ( _first || _previous_mass_h != _smvalues ->mh );
294 
295  bool new_gridParameters = (
296  _first
297  || new_mh
298  || _previous_f_B != _effvalues->fb
299  || _previous_f_BB != _effvalues->fbb
300  || _previous_f_WW != _effvalues->fww
301  || _previous_f_W != _effvalues->fw
302  || _previous_f_Phi_2 != _effvalues->fp2
303  );
304 
305  if ( _first ) _first = false;
306 
307  if ( new_mh ) _previous_mass_h = _smvalues ->mh;
308 
309  if ( new_gridParameters ) {
310 
311  _previous_f_B = _effvalues->fb;
312  _previous_f_W = _effvalues->fw;
313  _previous_f_BB = _effvalues->fbb;
314  _previous_f_WW = _effvalues->fww;
315  _previous_f_Phi_2 = _effvalues->fp2;
316 
317  }
318 
319  _Delta_g1_gamma = HDim6::d_g1_y ( _smvalues, _effvalues );
320  _Delta_g1_Z = HDim6::d_g1_z ( _smvalues, _effvalues );
321 
322  _Delta_kappa_gamma = HDim6::d_kappa_y( _smvalues, _effvalues );
323  _Delta_kappa_Z = HDim6::d_kappa_z( _smvalues, _effvalues );
324 
325  _Delta_g1_gaga = HDim6::d_g1_yy ( _smvalues, _effvalues );
326  _Delta_g1_WW = HDim6::d_g1_ww ( _smvalues, _effvalues );
327  _Delta_g1_Zga = HDim6::d_g1_zy ( _smvalues, _effvalues );
328  _Delta_g1_ZZ = HDim6::d_g1_zz ( _smvalues, _effvalues );
329 
330  _Delta_g2_WW = HDim6::d_g2_ww ( _smvalues, _effvalues );
331  _Delta_g2_gaga = HDim6::d_g2_yy ( _smvalues, _effvalues );
332  _Delta_g2_Zga = HDim6::d_g2_zy ( _smvalues, _effvalues );
333  _Delta_g2_ZZ = HDim6::d_g2_zz ( _smvalues, _effvalues );
334 
335  _s_paramater = s_parameter( _effvalues, _smvalues, GetInput( "Cutoff" ) );
336  _t_paramater = t_parameter( _effvalues, _smvalues, GetInput( "Cutoff" ) );
337  _u_parameter = u_parameter( _effvalues, _smvalues, GetInput( "Cutoff" ) );
338 
339  double error, chi2;
340 
341  if ( new_mh ) {
342 
343  hglgl_( _smvalues, _effsmvalues, &_SM_Gamma_hgg, &error );
344  hgaga_( _smvalues, _effsmvalues, &_SM_Gamma_hgaga, &error );
345  hgaz_ ( _smvalues, _effsmvalues, &_SM_Gamma_hZga, &error );
346  hmumu_( _smvalues, _effsmvalues, &_SM_Gamma_hmumu, &error );
347  htata_( _smvalues, _effsmvalues, &_SM_Gamma_htautau, &error );
348  hchch_( _smvalues, _effsmvalues, &_SM_Gamma_hcc, &error );
349  hstst_( _smvalues, _effsmvalues, &_SM_Gamma_hss, &error );
350  hbobo_( _smvalues, _effsmvalues, &_SM_Gamma_hbb, &error );
351 
352  }
353 
354  hglgl_( _smvalues, _effvalues, &_Gamma_hgg, &error );
355  hgaga_( _smvalues, _effvalues, &_Gamma_hgaga, &error );
356  hgaz_ ( _smvalues, _effvalues, &_Gamma_hZga, &error );
357  hmumu_( _smvalues, _effvalues, &_Gamma_hmumu, &error );
358  htata_( _smvalues, _effvalues, &_Gamma_htautau, &error );
359  hchch_( _smvalues, _effvalues, &_Gamma_hcc, &error );
360  hstst_( _smvalues, _effvalues, &_Gamma_hss, &error );
361  hbobo_( _smvalues, _effvalues, &_Gamma_hbb, &error );
362 
363  _normSM_Gamma_hgg = _Gamma_hgg / _SM_Gamma_hgg;
364  _normSM_Gamma_hgaga = _Gamma_hgaga / _SM_Gamma_hgaga;
365  _normSM_Gamma_hZga = _Gamma_hZga / _SM_Gamma_hZga;
366  _normSM_Gamma_hmumu = _Gamma_hmumu / _SM_Gamma_hmumu;
367  _normSM_Gamma_htautau = _Gamma_htautau / _SM_Gamma_htautau;
368  _normSM_Gamma_hcc = _Gamma_hcc / _SM_Gamma_hcc;
369  _normSM_Gamma_hss = _Gamma_hss / _SM_Gamma_hss;
370  _normSM_Gamma_hbb = _Gamma_hbb / _SM_Gamma_hbb;
371 
372  if ( new_gridParameters ) {
373 
374  if ( new_mh ) {
375 
376  if ( _calculate_Gamma_hZZ ) hzz_( _smvalues, _effsmvalues, &_SM_Gamma_hZZ, &error, &chi2 );
377  if ( _calculate_Gamma_hWW ) hww_( _smvalues, _effsmvalues, &_SM_Gamma_hWW, &error, &chi2 );
378 
379  }
380 
381  if ( _calculate_Gamma_hZZ ) hzz_( _smvalues, _effvalues, &_Gamma_hZZ, &_error_Gamma_hZZ, &_chi2_Gamma_hZZ );
382  if ( _calculate_Gamma_hWW ) hww_( _smvalues, _effvalues, &_Gamma_hWW, &_error_Gamma_hWW, &_chi2_Gamma_hWW );
383 
384  if ( _calculate_Gamma_hZZ ) _normSM_Gamma_hZZ = _Gamma_hZZ / _SM_Gamma_hZZ;
385  if ( _calculate_Gamma_hWW ) _normSM_Gamma_hWW = _Gamma_hWW / _SM_Gamma_hWW;
386 
387  }
388 
389  for (unsigned int iEnergy = 0; iEnergy < _energies.size(); ++iEnergy) {
390 
391  _smvalues->s = TMath::Power(_energies.at( iEnergy ), 2);
392 
393  k_ggh_(_smvalues, _effvalues, &_normSM_xs_ggh.at( iEnergy ), &error, &chi2);
394  ratio_bb_h_(_smvalues, _effvalues, &_normSM_xs_bbh.at( iEnergy ), &error, &chi2);
395  ratio_tth_(_smvalues, _effvalues, &_normSM_xs_tth.at( iEnergy ), &error, &chi2);
396  ratio_bg_bh_(_smvalues, _effvalues, &_normSM_xs_bh.at( iEnergy ), &error, &chi2);
397 
398  if (new_gridParameters) {
399 
400  if (new_mh) {
401 
402  if (_calculate_xs_Wh) HWRadiation_(_smvalues, _effsmvalues, &_SM_xs_Wh.at( iEnergy), &error, &chi2);
403  if (_calculate_xs_Zh) HZRadiation_(_smvalues, _effsmvalues, &_SM_xs_Zh.at( iEnergy ), &error, &chi2);
404  if (_calculate_xs_qqh_2flavor) ud_jjh_(_smvalues, _effsmvalues, &_SM_xs_qqh_2flavor.at( iEnergy ), &error, &chi2);
405  if (_calculate_xs_qqh_5flavor) udcsb_jjh_(_smvalues, _effsmvalues, &_SM_xs_qqh_5flavor.at( iEnergy ), &error, &chi2);
406 
407  }
408 
409  if (_calculate_xs_Wh) HWRadiation_(_smvalues, _effvalues, &_xs_Wh.at( iEnergy ), &_error_xs_Wh.at( iEnergy ), &_chi2_xs_Wh.at( iEnergy ) );
410  if (_calculate_xs_Zh) HZRadiation_(_smvalues, _effvalues, &_xs_Zh.at( iEnergy ), &_error_xs_Zh.at( iEnergy ), &_chi2_xs_Zh.at( iEnergy ) );
411  if (_calculate_xs_qqh_2flavor) ud_jjh_(_smvalues, _effvalues, &_xs_qqh_2flavor.at( iEnergy ), &_error_xs_qqh_2flavor.at( iEnergy ), &_chi2_xs_qqh_2flavor.at( iEnergy ) );
412  if (_calculate_xs_qqh_5flavor) udcsb_jjh_(_smvalues, _effvalues, &_xs_qqh_5flavor.at( iEnergy ), &_error_xs_qqh_5flavor.at( iEnergy ), &_chi2_xs_qqh_5flavor.at( iEnergy ) );
413 
414  if (_calculate_xs_Wh) _normSM_xs_Wh.at( iEnergy ) = _xs_Wh.at( iEnergy ) / _SM_xs_Wh.at( iEnergy );
415  if (_calculate_xs_Zh) _normSM_xs_Zh.at( iEnergy ) = _xs_Zh.at( iEnergy ) / _SM_xs_Zh.at( iEnergy );
416  if (_calculate_xs_qqh_2flavor) _normSM_xs_qqh_2flavor.at( iEnergy ) = _xs_qqh_2flavor.at( iEnergy ) / _SM_xs_qqh_2flavor.at( iEnergy );
417  if (_calculate_xs_qqh_5flavor) _normSM_xs_qqh_5flavor.at( iEnergy ) = _xs_qqh_5flavor.at( iEnergy ) / _SM_xs_qqh_5flavor.at( iEnergy );
418 
419  }
420 
421  }
