105 void Fittino::HerwigppCalculator::ReadLogFile() {
106 
107  unsigned int eventNumber = 1;
108  std::string matrixElement;
109  int colliding1;
110  int colliding2;
111  int incoming1;
112  int incoming2;
113  int intermediate;
114  int outgoing1;
115  int outgoing2;
116 
117  TDirectory* pwd = gDirectory;
118  TFile* rootFile = new TFile( "MatrixElements.root", "RECREATE" );
119  TTree* tree = new TTree( "Tree", "Tree" );
120  tree->Branch( "EventNumber", &eventNumber );
121  tree->Branch( "MatrixElement", &matrixElement );
122  tree->Branch( "Incoming1", &incoming1 );
123  tree->Branch( "Incoming2", &incoming2 );
124  tree->Branch( "Intermediate", &intermediate );
125  tree->Branch( "Outgoing1", &outgoing1 );
126  tree->Branch( "Outgoing2", &outgoing2 );
127 
128  std::ifstream logStream( _logFile );
129 
130  if ( !logStream.is_open() ) throw LogicException( "Can't open " + _logFile + "." );
131 
132  while( GetLine( logStream ) ) {
133 
134  if ( _line != "******************************************************************************" ) continue;
135 
136  GetLine( logStream );
137  if( _words.size() != 8 || _words[0] != "Event" || _words[1]!="number" || _words[5] != "performed" || _words[6] != "by" ) throw LogicException( "Unexpected line, expected event number!" );
138  unsigned int thisEventNumber = std::stoi( _words[2] );
139  if( thisEventNumber == eventNumber + 1 ) tree->Fill();
140  else if ( thisEventNumber != eventNumber ) throw LogicException( "Event missing!" );
141  eventNumber = thisEventNumber;
142 
143  GetLine( logStream );
144  if( _line != "==============================================================================") throw LogicException( "Unexpected line, expected seperating line!" );
145 
146  GetLine( logStream );
147  if( _line != "" ) throw LogicException( "Unexpected line, expected empty line.!" );
148 
149  GetLine( logStream );
150  if( _line != "--- Colliding particles:" ) throw LogicException( "Unexpected line, expected colliding particles.!" );
151  colliding1 = GetPDGID( logStream, 1 );
152  colliding2 = GetPDGID( logStream, 2 );
153 
154  GetLine( logStream );
155  if( _line != "------------------------------------------------------------------------------" ) throw LogicException( "Unexpected line, expected seperating line." );
156 
157  GetLine( logStream );
158  if ( _words.size() !=5 || _words[0] != "Primary" || _words[1] != "sub-process" || _words[2] != "performed" || _words[3] != "by" ) {
159 
160  throw LogicException( "Unexpected line, expected primary sub-process.!" );
161 
162  }
163  matrixElement = _words[4];
164 
165  GetLine( logStream );
166  if( _line != "--- incoming:" ) throw LogicException( "Unexpected line, expected incoming particles.!" );
167  incoming1 = GetPDGID( logStream, 3 );
168  incoming2 = GetPDGID( logStream, 4 );
169 
170  GetLine( logStream );
171  if( _line != "--- intermediates:" ) throw LogicException( "Unexpected line, expected intermediate particles.!" );
172  intermediate = GetPDGID( logStream, 5 );
173 
174  GetLine( logStream );
175  if( _line != "--- outgoing:" ) throw LogicException( "Unexpected line, expected outgoing particles.!" );
176  outgoing1 = GetPDGID( logStream, 6 );
177  outgoing2 = GetPDGID( logStream, 7 );
178 
179  GetLine( logStream );
180  if( _line != "------------------------------------------------------------------------------" ) throw LogicException( "Unexpected line, expected seperating line!" );
181 
182  }
183 
184  if ( eventNumber > 0 ) tree->Fill();
185 
186  tree->Write();
187  rootFile->Close();
188  delete rootFile;
189  pwd->cd();
190 
191 }

193 void Fittino::HerwigppCalculator::GetCrossSectionAndUncertainty( std::string input, double& xs, double& error ) {
194 
195  std::vector<std::string> significandErrorExponent;
196  split( significandErrorExponent, input, boost::is_any_of("()"), boost::token_compress_on );
197 
198  std::vector<std::string> decimal;
199  split( decimal, significandErrorExponent[0], boost::is_any_of("."), boost::token_compress_on );
200 
201  int numberOfDecimalPlaces;
202  if ( decimal.size() == 1 ) numberOfDecimalPlaces = 0;
203  else if ( decimal.size() == 2 ) numberOfDecimalPlaces = decimal.at(1).length();
204  else throw LogicException( "Determination of number of decimal places failed." );
205 
206  xs = std::stod( significandErrorExponent[0] + significandErrorExponent[2] );
207  error = std::stod( significandErrorExponent[1] + significandErrorExponent[2] ) * TMath::Power( 10, -1 * numberOfDecimalPlaces );
208 
209 }
