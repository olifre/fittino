146 double Fittino::HiggsBoundsHECCalculator::CalculateGammahTotal( double mass_h,
147  double g2_hiss_s,
148  double g2_hicc_s,
149  double g2_hibb_s,
150  double g2_hitt_s,
151  double g2_himumu_s,
152  double g2_hitautau_s,
153  double g2_hiWW,
154  double g2_hiZZ,
155  double g2_hiZga,
156  double g2_higaga,
157  double g2_higg,
158  double Gamma_hInvisible ) {
159 
160  double Gamma_hTotal = smgamma_h_( &mass_h )
161  * ( 1
162  + ( g2_hiss_s - 1 ) * smbr_hss_( &mass_h )
163  + ( g2_hicc_s - 1 ) * smbr_hcc_( &mass_h )
164  + ( g2_hibb_s - 1 ) * smbr_hbb_( &mass_h )
165  + ( g2_hitt_s - 1 ) * smbr_htoptop_( &mass_h )
166  + ( g2_himumu_s - 1 ) * smbr_hmumu_( &mass_h )
167  + ( g2_hitautau_s - 1 ) * smbr_htautau_( &mass_h )
168  + ( g2_hiWW - 1 ) * smbr_hww_( &mass_h )
169  + ( g2_hiZZ - 1 ) * smbr_hzz_( &mass_h )
170  + ( g2_hiZga - 1 ) * smbr_hzgam_( &mass_h )
171  + ( g2_higaga - 1 ) * smbr_hgamgam_( &mass_h )
172  + ( g2_higg - 1 ) * smbr_hgg_( &mass_h ) )
173  + Gamma_hInvisible;
174 
175  return Gamma_hTotal;
176 
177 }
