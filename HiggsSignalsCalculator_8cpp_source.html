87  int nHzero = _names_h0.size();

92  bool changed_whichAnalyses = ( whichAnalyses != _initialized_whichAnalyses );
93 
94  bool finishHB_becauseOfRunningHB = _HBisInitialized && _runHiggsBounds && ( changed_nHiggs || changed_whichAnalyses );
95  bool finishHB_becauseOfRunningHS = _HBisInitialized && _runHiggsSignals && ( changed_nHiggs || ( _HSisInitialized && changed_ExpData ) );
96 
97  bool finishHS_becauseOfRunningHS = _HSisInitialized && _runHiggsSignals && ( changed_nHiggs || changed_ExpData );
98  bool finishHS_becauseOfRunningHB = _HSisInitialized && _runHiggsBounds && changed_nHiggs;
99 
100  if ( finishHB_becauseOfRunningHB || finishHB_becauseOfRunningHS ) {
101 
102  finish_higgsbounds_();
103  _HBisInitialized = false;

105  }
106

112  }
113

136  _HSisInitialized = true;
