32 Fittino::HistogramMakerBase::HistogramMakerBase( ModelBase* model, const boost::property_tree::ptree& ptree )
33  : Tool( model, ptree ),
34  _iEntry( 0 ),
35  _globalAxis( new TGaxis() ) {
36 
37  int difference = _model->GetCollectionOfQuantities().At( 0 )->GetUpperBound() - _model->GetCollectionOfQuantities().At( 0 )->GetLowerBound();
38 
39  _numberOfPointsToProcess = difference;
40  _iEntry = _model->GetCollectionOfQuantities().At( 0 )->GetLowerBound();
41 
42  _globalAxis->SetMaxDigits( ptree.get<int>( "AxisMaxDigits", 3 ) );
43 
44  // Loop over the quantities scheduled in the Fittino input file and fill the histogram
45  // information into appropriate lists for later use.
46 
47  BOOST_FOREACH( const boost::property_tree::ptree::value_type & node, ptree ) {
48 
49  if ( node.first == "Histogram" ) {
50 
51  _quantityName.push_back( node.second.get<std::string> ( "Name" ) );
52  _plotName.push_back ( node.second.get<std::string> ( "PlotName" ) );
53  _numberOfBins.push_back( node.second.get<unsigned int>( "NumberOfBins" ) );
54  _logScale.push_back ( node.second.get<bool> ( "LogScale", false ) );
55 
56  // Depending on whether or not LogScale is set, the histogram upper and lower bounds
57  // are recomputed.
58 
59  if ( node.second.get<bool>( "LogScale", false ) ) {
60 
61  if ( node.second.get<double>( "LowerBound" ) <= 0. || node.second.get<double>( "UpperBound" ) <= 0. ) {
62 
63  throw ConfigurationException( "Histogram axis set to log scale but boundaries are not positive." );
64 
65  }
66  else {
67 
68  _lowerBound.push_back( TMath::Log10( node.second.get<double>( "LowerBound" ) ) );
69  _upperBound.push_back( TMath::Log10( node.second.get<double>( "UpperBound" ) ) );
70 
71  }
72 
73  }
74  else {
75 
76  _lowerBound.push_back( node.second.get<double>( "LowerBound" ) );
77  _upperBound.push_back( node.second.get<double>( "UpperBound" ) );
78 
79  }
80 
81  }
82 
83  }
84 
85  // Detemine the quantity index of each scheduled quantity in the list of model quantities. This
86  // is done because lookup via index is usually faster than lookup via name.
87 
88  for ( unsigned int iScheduledQuantity = 0; iScheduledQuantity < _quantityName.size(); ++iScheduledQuantity ) {
89 
90  for ( unsigned int iQuantity = 0; iQuantity < _model->GetCollectionOfQuantities().GetNumberOfElements(); ++iQuantity ) {
91 
92  if ( _quantityName[iScheduledQuantity] == _model->GetCollectionOfQuantities().At( iQuantity )->GetName() ) {
93 
94  _quantityIndex.push_back( iQuantity );
95 
96  }
97 
98  }
99 
100  }
101 
102  // Set the plotter.
103 
104  const Factory factory;
105 
106  const boost::property_tree::ptree::value_type& plotterNode = *( ptree.get_child( "Plotter" ).begin() );
107  std::string plotterType = plotterNode.first;
108  const boost::property_tree::ptree& plotterTree = plotterNode.second;
109  _plotter = factory.CreatePlotter( plotterType, _histogramVector, plotterTree );
