132 void Fittino::LHCLimitCalculator::CalculatePredictions() {
133 
134  std::map<std::string, std::vector<double> > parameterValueMap;
135 
136  // loop over the analyses
137  for ( std::map<std::string, std::vector<std::string> >::const_iterator itr = _relevantParametersMap.begin(); itr != _relevantParametersMap.end(); ++itr ) {
138 
139  std::vector<double> parameterValues;
140  bool isInListOfParameters = false;
141  bool isInListOfObservables = false;
142  // now loop over the parameters:
143  // try to find model parameter values in the list of parameters from the model and fill a vector with the values;
144 
145  for( int j = 0; j < (*itr).second.size(); ++j ) {
146 
147  for( int i = 0; i < _model->GetCollectionOfParameters().GetNumberOfElements(); ++i ) {
148  if( (*itr).second.at(j) == _model->GetCollectionOfParameters().At(i)->GetName() ) {
149  parameterValues.push_back( _model->GetCollectionOfParameters().At(i)->GetValue() );
150  isInListOfParameters = true;
151  break;
152  }
153  }
154 
155  // if not found in the list of parameters, try to find it in the list of observables...for now
156  if( !isInListOfParameters ) {
157 
158  for( int i = 0; i < _model->GetCollectionOfPredictions().GetNumberOfElements(); ++i ) {
159  if( (*itr).second.at(j) == _model->GetCollectionOfPredictions().At(i)->GetName() ) {
160  parameterValues.push_back( _model->GetCollectionOfPredictions().At(i)->GetValue() );
161  isInListOfObservables = true;
162  break;
163  }
164  }
165 
166  }
167 
168  if( !isInListOfParameters && !isInListOfObservables ) {
169 
170  throw ConfigurationException( "In LHCChi2Calculator: Won't be able to set values for relevant parameter in LHC Chi2 Contribution" );
171 
172  }
173 
174  }
175 
176  // loops complete. all parameters should have been filled.
177  if( parameterValues.size() != (*itr).second.size() ) {
178 
179  throw ConfigurationException( "In LHCChi2Calculator: Number of assigned parameters does not match number of relevant parameters! Something is wrong!" );
180 
181  }
182  // if found, add it:
183  parameterValueMap.insert( std::pair<std::string, std::vector<double> >( (*itr).first, parameterValues ) );
184 
185 
186  }
187 
188 
189  for( std::map<std::string, TH1D*>::const_iterator itr = _chi2Histograms1D.begin(); itr != _chi2Histograms1D.end(); ++itr ) {
190  _chi2Values.at( (*itr).first ) = (*itr).second -> Interpolate( parameterValueMap.at((*itr).first).at(0) );
191  }
192 
193  for( std::map<std::string, TH2D*>::const_iterator itr = _chi2Histograms2D.begin(); itr != _chi2Histograms2D.end(); ++itr ) {
194  _chi2Values.at( (*itr).first ) = (*itr).second -> Interpolate( parameterValueMap.at((*itr).first).at(0), parameterValueMap.at((*itr).first).at(1) );
195  }
196 
197  for( std::map<std::string, TH3D*>::const_iterator itr = _chi2Histograms3D.begin(); itr != _chi2Histograms3D.end(); ++itr ) {
198 
199  _chi2Values.at( (*itr).first ) = (*itr).second -> Interpolate( parameterValueMap.at((*itr).first).at(0), parameterValueMap.at((*itr).first).at(1), parameterValueMap.at((*itr).first).at(2) );
200  }
201 
202  for( std::map<std::string, THnSparseD*>::const_iterator itr = _chi2HistogramsnD.begin(); itr != _chi2HistogramsnD.end(); ++itr ) {
203  _chi2Values.at( (*itr).first ) = InterpolateND( (*itr).second, parameterValueMap.at((*itr).first) );
204 
205  }

242 double Fittino::LHCLimitCalculator::InterpolateND( THnSparse *histogram, std::vector<double> parameterValues ) {
243 
244  // first step : find the 2^n nearest bins
245  std::vector<double> closestHigherValues;
246  std::vector<double> closestLowerValues;
247 
248  for( unsigned int i = 0; i < parameterValues.size(); ++i ) {
249 
250  bool underFlow = false;
251  double maxValue = histogram->GetAxis( i )->GetBinCenter( histogram->GetAxis( i )->GetNbins() );
252  double minValue = histogram->GetAxis( i )->GetBinCenter( 1 );
253  if( parameterValues.at( i ) > maxValue ) {
254  parameterValues.at( i ) = maxValue;
255  }
256  if( parameterValues.at( i ) <= minValue ) {
257  parameterValues.at( i ) = minValue;
258  underFlow = true;
259  }
260 
261  int binIndex = histogram->GetAxis( i )->FindFixBin( parameterValues.at( i ) );
262  double binCenter = histogram->GetAxis( i )->GetBinCenter( binIndex );
263 
264  if( binCenter >= parameterValues.at( i ) && !underFlow ) {
265  closestHigherValues.push_back( histogram->GetAxis( i )->GetBinCenter( binIndex ) );
266  closestLowerValues.push_back( histogram->GetAxis( i )->GetBinCenter( binIndex - 1 ) );
267  }
268  else {
269  closestHigherValues.push_back( histogram->GetAxis( i )->GetBinCenter( binIndex + 1 ) );
270  closestLowerValues.push_back( histogram->GetAxis( i )->GetBinCenter( binIndex ) );
271  }
272 
273  }
274 
275  // got the reference points
276  // now get and store the function values at the reference poitns;
277 
278  std::vector<double> referenceValues;
279  std::vector<double> referenceValuesTemp;
280  std::vector<std::vector<double> > binaryLowerHigher;
281  binaryLowerHigher.push_back( closestLowerValues );
282  binaryLowerHigher.push_back( closestHigherValues );
283  std::vector<int> position( closestHigherValues.size(), 1 );
284  std::vector<int> coordinates( closestHigherValues.size(), 1 );
285 
286  // loop over the 2^n points like this and fill the vector referenceValues like this:
287  //
288  // for each coordinate, have a 'boolean': 0 means the next lowest value, 1 means the next highest value
289  // then store the referenceValues in the following order:
290  // 0 0 0 0 . . .
291  // 1 0 0 0 . . .
292  // 0 1 0 0 . . .
293  // 1 1 0 0 . . .
294  // 0 0 1 0 . . .
295  // 1 0 1 0 . . .
296  // 0 1 1 0 . . .
297  // 1 1 1 0 . . .
298  // 0 0 0 1 . . .
299  // so the boolean for the nth parameter switches value every 2^(n-1)th time
300 
301  for( unsigned int i = 0; i < pow( 2, closestHigherValues.size() ); ++i ) {
302  for( unsigned int j = 0; j < closestHigherValues.size(); ++j ) {
303  int period = pow( 2, j );
304  if( i % period == 0 ) {
305  position.at( j ) = ( position.at( j ) + 1 ) % 2;
306  }
307  coordinates.at( j ) = histogram->GetAxis( j )->FindFixBin( binaryLowerHigher.at( position.at( j ) ).at( j ) );
308  }
309  referenceValues.push_back( histogram->GetBinContent( &coordinates.at( 0 ) ) );
310  }
311 
312 
313  // reference values are stored. now interpolate iteratively all coordinates:
314  // combine the ith and (i+1)th entry from the vector referenceValues to the interpolated value for the nth dimension, until
315  // the n equals the number of dimensions from the input histogram and the new referenceValues has only 1 entry left.
316 
317  int interpolateCoordinate = 0;
318 
319  while( interpolateCoordinate < closestHigherValues.size() ) {
320  referenceValuesTemp.clear();
321  for( unsigned int i = 0; i < referenceValues.size(); ++i ) {
322  double return_inter = referenceValues.at( i ) + ( referenceValues.at( i ) - referenceValues.at( ++i ) ) / ( closestLowerValues.at( interpolateCoordinate ) - closestHigherValues.at( interpolateCoordinate ) ) * ( parameterValues.at( interpolateCoordinate ) - closestLowerValues.at( interpolateCoordinate ) );
323  referenceValuesTemp.push_back( return_inter );
324  }
325 
326  // clear referenceValues and re-fill it with the interpolated values
327  referenceValues.clear();
328  for( unsigned int i = 0; i < referenceValuesTemp.size(); ++i ) {
329  referenceValues.push_back( referenceValuesTemp.at( i ) );
330  }
331  interpolateCoordinate += 1;
332  }
333 
334  // all done. return interplated value:
335  return referenceValues.at( 0 );
336 
337 }
