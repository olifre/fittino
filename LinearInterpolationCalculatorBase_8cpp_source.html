157 void Fittino::LinearInterpolationCalculatorBase::InterpolateND() {
158 
159  // first step : find the nearest bins
160 
161  std::vector<int> closestHigherBinIndices( _nDimensions );
162  std::vector<int> closestLowerBinIndices( _nDimensions );
163 
164  for( unsigned int iDimension = 0; iDimension < _nDimensions; ++iDimension) {
165 
166  bool underFlow = false;
167  double maxValue = _histogramnD->GetAxis( iDimension )->GetBinCenter( _histogramnD->GetAxis( iDimension )->GetNbins() );
168  double minValue = _histogramnD->GetAxis( iDimension )->GetBinCenter( 1 );
169 
170  if( _interpolationPoint.at( iDimension ) > maxValue ) {
171 
172  _interpolationPoint.at( iDimension ) = maxValue;
173 
174  }
175  if( _interpolationPoint.at( iDimension ) <= minValue ) {
176 
177  _interpolationPoint.at( iDimension ) = minValue;
178  underFlow = true;
179 
180  }
181 
182  int binIndex = _histogramnD->GetAxis( iDimension )->FindFixBin( _interpolationPoint.at( iDimension ) );
183  double binCenter = _histogramnD->GetAxis( iDimension )->GetBinCenter( binIndex );
184 
185  if( binCenter >= _interpolationPoint.at( iDimension ) && !underFlow ) {
186 
187  closestHigherBinIndices.at( iDimension ) = binIndex;
188  closestLowerBinIndices.at( iDimension ) = binIndex - 1;
189 
190  }
191  else {
192 
193  closestHigherBinIndices.at( iDimension ) = binIndex + 1 ;
194  closestLowerBinIndices.at( iDimension ) = binIndex;
195 
196  }
197 
198  }
199 
200  // now get and store the function values at the reference points;
201  // loop over the 2^n points like this and fill the vector referenceValues like this:
202  //
203  // for each coordinate, have a 'boolean': 0 means the next lowest value, 1 means the next highest value
204  // then store the referenceValues in the following order:
205  // 0 0 0 0 . . .
206  // 1 0 0 0 . . .
207  // 0 1 0 0 . . .
208  // 1 1 0 0 . . .
209  // 0 0 1 0 . . .
210  // 1 0 1 0 . . .
211  // 0 1 1 0 . . .
212  // 1 1 1 0 . . .
213  // 0 0 0 1 . . .
214 
215  unsigned int nReferencePoints = TMath::Power( 2, static_cast<int>( _nDimensions ) );
216  std::vector<double> referenceFunctionValues( nReferencePoints );
217 
218  for( unsigned int iReferencePoint = 0; iReferencePoint < nReferencePoints; ++iReferencePoint) {
219 
220  std::bitset<100> bits(iReferencePoint);
221  std::vector<int> binIndices;
222 
223  for( unsigned int iDimension = 0; iDimension < _nDimensions; ++iDimension ) {
224 
225  int binIndex;
226 
227  if ( bits.test( iDimension ) ) {
228 
229  binIndex = closestHigherBinIndices.at( iDimension );
230 
231  }
232  else {
233 
234  binIndex = closestLowerBinIndices.at( iDimension );
235 
236  }
237 
238  binIndices.push_back( binIndex );
239 
240  }
241 
242  referenceFunctionValues.at( iReferencePoint ) = _histogramnD->GetBinContent ( &binIndices.at(0) );
243 
244  }
245 
246  // reference values are stored. now interpolate iteratively all coordinates:
247  // combine the ith and (i+1)th entry from the vector referenceValues to the interpolated value for the nth dimension, until
248  // the n equals the number of dimensions from the input histogram and the new referenceValues has only 1 entry left.
249 
250  for ( unsigned int iDimension = 0; iDimension < _nDimensions; iDimension++ ) {
251 
252  std::vector<double> referenceValuesTemp;
253 
254  double x = _interpolationPoint.at( iDimension );
255  double x0 = _histogramnD->GetAxis( iDimension )->GetBinCenter( closestLowerBinIndices.at( iDimension ) );
256  double x1 = _histogramnD->GetAxis( iDimension )->GetBinCenter( closestHigherBinIndices.at( iDimension ) );
257 
258  for( unsigned int i = 0; i < referenceFunctionValues.size(); i += 2 ) {
259 
260  double y0 = referenceFunctionValues.at( i );
261  double y1 = referenceFunctionValues.at( i + 1 );
262 
263  double y = y0 + ( y1 - y0 ) / ( x1 - x0 ) * ( x - x0 );
264 
265  referenceValuesTemp.push_back( y );
266 
267  }
268 
269  referenceFunctionValues = referenceValuesTemp;
270 
271  }
272 
273  _value = referenceFunctionValues.at( 0 );
274 
275 }
