130 bool Fittino::MarkovChainSampler::IsAccepted() {
131 
132  if ( !IsInBounds() ) return false;
133 
134  double DeltaChi2 = _chi2 - _chi2OfLastAcceptedPoint;
135  double rho = exp( -DeltaChi2 / 2. );
136  double randomThreshold = _randomGenerator->Uniform( 1. );
137 
138  if ( rho < randomThreshold ) return false;
139 
140  return true;
