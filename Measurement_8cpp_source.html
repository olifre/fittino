76 void Fittino::Measurement::Update() {
77 
78  _prediction->Update();
79 
80  _totalUncertainty = 0;
81 
82  std::vector< Quantity* >::iterator it;
83 
84  for ( it = _uncertainties.begin(); it != _uncertainties.end(); ++it ) {
85 
86  (*it)->Update();
87  _totalUncertainty += TMath::Power( (*it)->GetValue(), 2 );
88 
89  }
90 
91  _totalUncertainty = TMath::Sqrt( _totalUncertainty );
92 
93  _deviation = _prediction->GetValue() - _measurement->GetValue();
94 
95  if ( ( _type == Type::LowerLimit && _deviation > 0 ) || ( _type == Type::UpperLimit && _deviation < 0 ) ||
96  ( _type == Type::Box && _deviation <= _totalUncertainty ) ) {
97 
98  _deviation = 0;
99 
100  }
101  else if( _type == Type::Box ) {
102 
103  _deviation = std::numeric_limits<double>::infinity();
104 
105  }
106 
107  _pull = _deviation / _totalUncertainty;
108 
109  if ( _totalUncertainty == 0 && _deviation == 0 ) {
110 
111  _pull = 0;
112 
113  }
114 
115  _chi2 = TMath::Power( _pull, 2 );
116 
117 }
