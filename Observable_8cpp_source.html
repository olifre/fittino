165 void Fittino::Observable::SmearMeasuredValue( ) {
166 
167  RandomGenerator* randomGenerator = Fittino::RandomGenerator::GetInstance();
168 
169  if( _smearingType == "Gaus" ) {
170 
171  _measuredValue = randomGenerator->Gaus( _bestFitPrediction, _measuredError );
172 
173  }
174  else if( _smearingType == "Poisson" ) {
175 
176  // we will have to think of a better way to do this. For now, this is for the ATLAS 0 Lepton Analysis.
177  // Smearing works in the following way:
178  // In the input file, set 1) NObs = Number of expected background events
179  // 2) NExpBestFit = Number of expected events at best fit point == > Number of expected signal events = NExpBestFit - NObs
180  // 3) MeasuredError1 = systematic uncertainty on the number of background events
181  // 4) MeausredError2 = systematic uncertainty on the number of signal events
182  _measuredValue = randomGenerator->Poisson( _measuredValue*randomGenerator->Gaus( 1., _error1) + (_bestFitPrediction - _measuredValue)*randomGenerator->Gaus(1., _error2 ));
183 
184  }
185  else {
186 
187  Messenger::GetInstance() << Messenger::ALWAYS << "Unkonwn SmearingType " << _smearingType << " for observable " << GetPrediction()->GetName() << ". Measured Value will not be smeared. " << Messenger::Endl;
188 
189  }
190 
191 }
