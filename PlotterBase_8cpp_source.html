245 void Fittino::PlotterBase::DrawLogo() {
246 
247  _fittinoLogo->SetConstRatio( 1 );
248  _fittinoLogo->SetImageQuality( TAttImage::kImgBest );
249  _fittinoLogo->SetImageCompression( 0 );
250 
251  const float canvasHeight = _canvas->GetWindowHeight();
252  const float canvasWidth = _canvas->GetWindowWidth();
253  const float canvasAspectRatio = canvasHeight / canvasWidth;
254  const float width = 0.19;
255  const float xLowerEdge = 0.02;
256  const float yLowerEdge = 0.853;
257  const float xUpperEdge = xLowerEdge + width;
258  const float yUpperEdge = yLowerEdge + width * _fittinoLogo->GetHeight() / _fittinoLogo->GetWidth() / canvasAspectRatio;
259 
260  TPad *fittinoLogoPad = new TPad( "fittinoLogoPad", "fittinoLogoPad", xLowerEdge, yLowerEdge, xUpperEdge, yUpperEdge );
261  fittinoLogoPad->Draw( "same" );
262  fittinoLogoPad->cd();
263  _fittinoLogo->Draw( "xxx" );
264  _canvas->cd();
