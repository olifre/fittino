74 void Fittino::SLHAeaSLHADataStorage::AddLine( const std::string& path ) {
75 
76  // This function creates a new SLHA line within an existing SLHA block. The individual data
77  // fields of this line, represented by a colon-seperated string named "path", are passed as the
78  // argument.
79 
80  // Create a new line.
81 
82  SLHAea::Line line;
83 
84  // Extract the name of the SLHA block, which should be the first entry in the list of
85  // colon-seperated strings.
86 
87  std::string blockName = path.substr( 0, path.find_first_of( ":" ) );
88 
89  // Copy the input path and strip the first entry. The next entry is then added to the line.
90  // Continue until no further colons are found.
91 
92  std::string tmpPath( path );
93 
94  while ( tmpPath.find_first_of( ":" ) <= tmpPath.length() ) {
95 
96  tmpPath.erase( 0, tmpPath.find_first_of( ":" ) + 1 );
97  line << tmpPath.substr( 0, tmpPath.find_first_of( ":" ) );
98 
99  }
100 
101  ( *_slhaeaDataStorage )[blockName][""] << line;
102 
103 }
