220 void Fittino::SPhenoSLHACalculator::CalculatePredictions() {
221 
222  _inputFile->Write();
223 
224  try {
225 
226  _returnValue = _executor->Execute();
227 
228  }
229  catch ( const TimeoutExecutorException& e ) {
230 
231  throw CalculatorException( _name, "Timeout" );
232 
233  }
234 
235  if ( _returnValue != 0 ) {
236 
237  throw CalculatorException( _name, "Return value " + boost::lexical_cast<std::string>( _returnValue ) );
238 
239  }
240 
241  _slhaOutputDataStorage->Clear();
242  try {
243 
244  _slhaOutputDataStorage->ReadFile( _slhaOutputFileName );
245 
246  }
247  catch ( const SLHAFileException& e ) {
248 
249  throw CalculatorException( _name, "Could not read SLHA File" );
250 
251  }
252 
253  for ( unsigned int i = 0; i < GetCollectionOfQuantities().GetNumberOfElements(); ++i ) {
254 
255  GetCollectionOfQuantities().At( i )->Update();
256 
257  }
258 
259 }
