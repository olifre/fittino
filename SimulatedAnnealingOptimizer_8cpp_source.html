52 void Fittino::SimulatedAnnealingOptimizer::UpdateModel() {
53 
54  std::vector<double> stepWidth;
55 
56  for ( unsigned int i = 0; i < _model->GetNumberOfParameters(); i++ ) {
57 
58  stepWidth.push_back( 1. );
59 
60  }
61 
62  // The candidate for the new model is taken as the best model out of a number of trials which is
63  // equal to 3 times the number of parameters of the model, but at least 60.
64 
65  unsigned int numberOfTrials = 60;
66 
67  if ( _model->GetNumberOfParameters() >= 20 ) {
68 
69  numberOfTrials = 3 * _model->GetNumberOfParameters();
70 
71  }
72 
73  // The actual best model and chi2 are stored here for future reference.
74 
75  ModelBase* bestModel = _model->Clone();
76  double bestChi2 = bestModel->GetChi2();
77 
78  for ( unsigned int i = 0; i < numberOfTrials; i++ ) {
79 
80  // Within every trial, each parameter is varied one after another and the number of accepted
81  // parameter variations is counted. A parameter variation is accepted if the new model has a
82  // better chi2 value or if it fulfills the Metropolis criterion. This is repeated 20 times.
83  // After that the step width for the next iteration of variations is updated for each
84  // parameter individually.
85 
86  std::vector<int> numberOfAcceptedVariations;
87 
88  // Reset of the counter.
89 
90  for ( unsigned int i = 0; i < _model->GetNumberOfParameters(); i++ ) {
91 
92  numberOfAcceptedVariations.push_back( 0 );
93 
94  }
95 
96  for ( unsigned int j = 0; j < 20; j++ ) {
97 
98  // Variation of one parameter at a time.
99 
100  for ( unsigned int k = 0; k < _model->GetNumberOfParameters(); k++ ) {
101 
102  ModelBase* oldModel = _model->Clone();
103  double oldChi2 = oldModel->GetChi2();
104 
105  // Update of the model.
106 
107  _model->GetCollectionOfParameters().At( k )->SetValue( _model->GetCollectionOfParameters().At( k )->GetValue() + _randomGenerator->Uniform( -1, 1 ) * stepWidth[k] );
108 
109  double newChi2 = _model->GetChi2();
110 
111  if ( newChi2 > oldChi2 ) {
112 
113  // This is the Metropolis criterion.
114 
115  double acceptance = exp( -( newChi2 - oldChi2 ) / _temperature );
116  double randomThreshold = _randomGenerator->Uniform( 0, 1 );
117 
118  if ( acceptance > randomThreshold ) {
119 
120  numberOfAcceptedVariations[k]++;
121 
122  }
123  else {
124 
125  _model = oldModel;
126 
127  }
128 
129  }
130  else {
131 
132  numberOfAcceptedVariations[k]++;
133 
134  // If the model has the lowest chi2 so far, the bestChi2 and bestModel are
135  // updated, respectively.
136 
137  if ( newChi2 < bestChi2 ) {
138 
139  bestChi2 = newChi2;
140  bestModel = _model->Clone();
141 
142  }
143 
144  }
145 
146  }
147 
148  }
149 
150  // Adjustment of the stepWidth.
151 
152  for ( unsigned int iParameter = 0; iParameter < _model->GetNumberOfParameters(); iParameter++ ) {
153 
154  double samplingEfficiency = static_cast<double>( numberOfAcceptedVariations[iParameter] ) / 20.;
155 
156  if ( samplingEfficiency > 0.6 ) {
157 
158  stepWidth[iParameter] *= 2. * ( samplingEfficiency - 0.6 ) / 0.4 + 1.;
159 
160  }
161  else if ( samplingEfficiency < 0.4 ) {
162 
163  stepWidth[iParameter] /= 2. * ( 0.4 - samplingEfficiency ) / 0.4 + 1.;
164 
165  }
166 
167  }
168 
169  }
170 
171  // Reduction of the temperature.
172 
173  _temperature *= _temperatureReductionFactor;
174 
175 }
