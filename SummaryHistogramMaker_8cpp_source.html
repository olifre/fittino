92 
93  if ( _logScale ) {

98

103  }

106 
107  if ( _spacing ) nybins = 2 * nybins - 1;
108

110

112  histogramName.c_str(),
113  _numberOfBins, xBins, nybins , 0., nybins );
114

116 
117  histogramName = "Summary1Sigma";
118 
119  _histogram1Sigma = new TH2D( histogramName.c_str(),
120  histogramName.c_str(),
121  _numberOfBins, xBins, nybins, 0., nybins );

152 
153  }

161

165  for ( unsigned int iQuantity = 0; iQuantity < _quantityName.size(); ++iQuantity ) {
166

168 
169  int bin = FindBinY( iQuantity );

172  double up = _histogram2Sigma->GetYaxis()->GetBinUpEdge( bin );
173  TGraph* graph = new TGraph( 2 );
174  graph->SetPoint( 0, bestFitValue, low );
175  graph->SetPoint( 1, bestFitValue, up );
176  _plotter->AddGraph( graph );
177 
178  }

222 void Fittino::SummaryHistogramMaker::Execute() {
223 
224  _iterationCounter = _model->GetCollectionOfParameters().At( 0 )->GetLowerBound();
225 
226  int upperBound = _model->GetCollectionOfParameters().At( 0 )->GetUpperBound();
227 
228  while ( _iterationCounter < upperBound ) {
229

241  }

248 
249  }

280 void Fittino::SummaryHistogramMaker::UpdateModel() {
281 
282  // For each quantity fill the bins corresponding to the 1 and 2 sigma regions.
283 
284  _model->GetCollectionOfParameters().At( 0 )->SetValue( _iterationCounter );
285 
286  double chi2 = _model->GetChi2();
287 
288  double normalizedChi2 = chi2 - _lowestChi2;
289 
290  // Loop over all quantities.
291 
292  for ( unsigned int iQuantity = 0; iQuantity < _quantityName.size(); ++iQuantity ) {
293 
294  int ybin = FindBinY( iQuantity );
295 
296  int bin = FindBin( _model->GetCollectionOfQuantities().At( _quantityIndex[iQuantity] )->GetValue() );
297 
298  if ( normalizedChi2 <= 1. ) {
299 
300  // If the bin corresponds to the 1 sigma region, fill with 1.2.
301 
302  _histogram1Sigma->SetBinContent( bin, ybin , 1.2 );
303 
304  }
305  else if ( normalizedChi2 <= 4. && normalizedChi2 > 1. ) {
306 
307  // If the bin corresponds to the 2 sigma region, fill with 0.8.
308 
309  _histogram2Sigma->SetBinContent( bin, ybin, 0.8 );
310 
311  }
312 
313  }
