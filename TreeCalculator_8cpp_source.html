100 void Fittino::TreeCalculator::AddPredictions( ) {
101 
102  TIterator *iter = _inputTree->GetIteratorOnAllLeaves();
103 
104  while ( TObject* obj = ( *iter )() ) {
105 
106  TLeaf *leaf = ( TLeaf* ) obj;
107 
108  bool excludeLeaf = false;
109 
110  for ( unsigned int j = 0; j < _excludedLeaves.size(); ++j ) {
111 
112  if ( !strcmp( leaf->GetName(), _excludedLeaves.at( j ).c_str() ) ) {
113 
114  excludeLeaf = true;
115 
116  }
117 
118  }
119 
120  if ( _excludeAllLeaves ) excludeLeaf = true;
121 
122  for ( unsigned int j = 0; j < _includedLeaves.size(); ++j ) {
123 
124  if ( !strcmp( leaf->GetName(), _includedLeaves.at( j ).c_str() ) ) {
125 
126  excludeLeaf = false;
127 
128  }
129 
130  }
131 
132  if ( excludeLeaf ) continue;
133 
134  if ( !strcmp( leaf->GetTypeName(), "Double_t" ) ) {
135 
136  _predictionMap.insert( std::make_pair( leaf->GetName(), 0. ) );
137  _inputTree->SetBranchStatus( leaf->GetName(), 1 ) ;
138  _inputTree->SetBranchAddress( leaf->GetName(), &_predictionMap.at( leaf->GetName() ) );
139 
140  bool exchangeLeaf = false;
141  std::string newName;
142  for( unsigned int j = 0; j < _exchangeLeaves.size(); ++j ) {
143 
144  if( !strcmp( leaf->GetName(), _exchangeLeaves.at(j).first.c_str() ) ) {
145 
146  exchangeLeaf = true;
147  newName = _exchangeLeaves.at(j).second;
148 
149  }
150  }
151 
152  if( exchangeLeaf ) {
153  std::cout << "replacing " << leaf->GetName() << " with " << newName << std::endl;
154  AddQuantity( new SimplePrediction( newName.c_str(), "", _predictionMap.at( leaf->GetName() ) ) );
155  }
156  else {
157  AddQuantity( new SimplePrediction( leaf->GetName(), "", _predictionMap.at( leaf->GetName() ) ) );
158  }
159 
160  }
161 
162  }
163 
164  _numberOfTreeIterations = _inputTree->GetEntries();
165  AddQuantity( new SimplePrediction( "TreeIterations", "", _numberOfTreeIterations ) );
166
